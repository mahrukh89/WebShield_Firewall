import os
import re

app = Flask(__name__)

# HTML template for the web interface

def run_command(cmd):
    """Run a shell command and return output."""
    try:
        result = subprocess.run(cmd, shell=True, capture_output=True, text=True, check=True)
        return result.stdout
    except subprocess.CalledProcessError as e:
        return f"Error: {e.stderr}"

def get_rules():
    """Get current iptables rules."""
    output = run_command("sudo iptables -L -n --line-numbers")
    rules = []
    for line in output.split('\n'):
        if re.match(r'^\d+', line):
            parts = line.split()
            if len(parts) > 2:
                chain = parts[0] if parts[0].isdigit() else "INPUT"  # Assuming INPUT chain
                rule = ' '.join(parts[1:])
                rules.append((chain, rule, parts[0]))
    return rules

def add_rule(protocol, src_ip, dst_ip, src_port, dst_port, action):
    """Add a packet filtering rule."""
    cmd = f"sudo iptables -A INPUT -p {protocol}"
    if src_ip:
        cmd += f" -s {src_ip}"
    if dst_ip:
 cmd += f" -d {dst_ip}"
    if src_port:
        cmd += f" --sport {src_port}"
    if dst_port:
        cmd += f" --dport {dst_port}"
    cmd += f" -j {action}"
    run_command(cmd)

def delete_rule(line_num):
    """Delete a rule by line number."""
    run_command(f"sudo iptables -D INPUT {line_num}")

def toggle_nat(enable):
    """Enable or disable NAT."""
    if enable:
        run_command("sudo sysctl -w net.ipv4.ip_forward=1")
        run_command("sudo iptables -t nat -A POSTROUTING -o eth0 -j MASQUERADE")  # Assuming eth0 is the outgoing interface
    else:
        run_command("sudo sysctl -w net.ipv4.ip_forward=0")
        run_command("sudo iptables -t nat -F POSTROUTING")

def get_logs():
    """Get recent firewall logs from kernel journal (Kali Linux)."""
    return run_command("sudo journalctl -k -n 20 | grep Firewall")


def get_monitoring():
    """Get packet counts."""
    return run_command("sudo iptables -L -n -v")
@app.route('/')
def index():
    rules = get_rules()
    logs = get_logs()
    monitoring = get_monitoring()
    return render_template_string(HTML_TEMPLATE, rules=rules, logs=logs, monitoring=monitoring)

@app.route('/add_rule', methods=['POST'])
def add_rule_route():
    protocol = request.form['protocol']
    src_ip = request.form.get('src_ip', '')
    dst_ip = request.form.get('dst_ip', '')
    src_port = request.form.get('src_port', '')
    dst_port = request.form.get('dst_port', '')
    action = request.form['action']
    add_rule(protocol, src_ip, dst_ip, src_port, dst_port, action)
    return redirect(url_for('index'))

@app.route('/delete_rule/<int:line_num>')
def delete_rule_route(line_num):
    delete_rule(line_num)
    return redirect(url_for('index'))

@app.route('/toggle_nat', methods=['POST'])
def toggle_nat_route():
    if 'enable' in request.form:
        toggle_nat(True)
    elif 'disable' in request.form:
        toggle_nat(False)
    return redirect(url_for('index'))

if __name__ == '__main__':
    # Ensure iptables is available and set basic stateful rules
    run_command("sudo iptables -F")  # Flush existing rules
    run_command("sudo iptables -A INPUT -m state --state ESTABLISHED,RELATED -j ACCEPT")  # Stateful rule
    run_command("sudo iptables -A INPUT -j LOG --log-prefix 'Firewall: '")  # Logging
    app.run(host='0.0.0.0', port=5000, debug=False)
